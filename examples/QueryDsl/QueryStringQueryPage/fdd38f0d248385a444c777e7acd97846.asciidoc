var searchResponse = client.Search<object>(s => s
    .Index("")
    .Query(q => q
        .QueryString(qs => qs
            .Fields(f => f
                .Field("title")
                .Field("content")
            )
            .Query("this OR that OR thus")
            .MinimumShouldMatch(2)
        )
    )
);
