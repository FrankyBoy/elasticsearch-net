var searchResponse = client.Search<Account>(s => s
    .Index("bank")
    .Query(q => q
        .Bool(b => b
            .Must(mu => mu
                .Match(m => m
                    .Field(f => f.Age)
                    .Query("40")
                )
            )
            .MustNot(mn => mn
                .Match(m => m
                    .Field(ff => ff.State)
                    .Query("ID")
                )
            )
        )
    )
);
