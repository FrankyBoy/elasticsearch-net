var searchResponse = client.Search<Account>(s => s
    .Index("bank")
    .Query(q => q
        .Bool(b => b
            .MustNot(mn => mn
                    .Match(m => m
                        .Field(f => f.Address)
                        .Query("mill")
                    ), mn => mn
                    .Match(m => m
                        .Field(f => f.Address)
                        .Query("lane")
                    )
            )
        )
    )
);
