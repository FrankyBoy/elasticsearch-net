var searchResponse = client.Search<Account>(s => s
    .Index("bank")
    .Query(q => q
        .Bool(b => b
            .Must(mu => mu
                .MatchAll()
            )
            .Filter(f => f
                .Range(r => r
                    .Field(ff => ff.Balance)
                    .GreaterThanOrEquals(20000)
                    .LessThanOrEquals(30000)
                )
            )
        )
    )
);
